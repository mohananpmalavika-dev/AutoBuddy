**Focus**: Taxi, XL, Traveller, Mini Truck, Truck

├── 6-seater (Innova, Ertiga)

└── Heavy Truck (5-7 ton)

#### 1.3 Dynamic Pricing Engine
```
Price = Base Rate + Distance Charge + Time Charge + Surge + Vehicle Premium

Vehicle Premiums:
- Auto: ₹0 (baseline)
- Taxi: +₹30
- XL: +₹100
- Mini Truck: +₹150
- Truck: +₹300

Surge Multipliers:
- Peak hours (8-10 AM, 5-7 PM): 1.5x
- Bad weather: 1.3x
- High demand zone: 1.2x
```

## 🗄️ DATABASE ARCHITECTURE

### Core New Tables Needed

```sql
-- Vehicle Types (Canonical)
CREATE TABLE vehicle_types (
  id UUID PRIMARY KEY,
  category ENUM('auto', 'taxi', 'traveller', 'truck', 'ambulance', 'school'),
  subcategory VARCHAR(50),
  name VARCHAR(100),
  capacity_pax INT,
  capacity_weight_kg INT,
  base_rate DECIMAL(10,2),
  per_km_rate DECIMAL(10,2),
  per_minute_rate DECIMAL(10,2),
  vehicle_insurance_req BOOLEAN,
  driver_training_req VARCHAR(100),
  compliance_docs JSONB,
  active BOOLEAN DEFAULT true,
  created_at TIMESTAMP
);

-- Ride Products (Per vehicle type)
CREATE TABLE ride_products (
  id UUID PRIMARY KEY,
  vehicle_type_id UUID REFERENCES vehicle_types,
  name VARCHAR(100),
  description TEXT,
  icon VARCHAR(50),
  display_priority INT,
  base_price DECIMAL(10,2),
  surge_enabled BOOLEAN,
  cancellation_policy JSONB,
  features JSONB,
  active BOOLEAN DEFAULT true,
  created_at TIMESTAMP
);

-- Corporate Contracts
CREATE TABLE corporate_contracts (
  id UUID PRIMARY KEY,
  company_id UUID,
  company_name VARCHAR(200),
  contract_value DECIMAL(12,2),
  monthly_billing DECIMAL(12,2),
  num_employees INT,
  approval_required BOOLEAN,
  budget_limit DECIMAL(12,2),
  contract_start DATE,
  contract_end DATE,
  status ENUM('active', 'pending', 'expired'),
  contact_person VARCHAR(100),
  contact_phone VARCHAR(20),
  created_at TIMESTAMP
);

-- Tourism Packages
CREATE TABLE tourism_packages (
  id UUID PRIMARY KEY,
  package_name VARCHAR(200),
  destination VARCHAR(100),
  duration_days INT,
  inclusions JSONB,
  itinerary JSONB,
  price_per_person DECIMAL(10,2),
  max_group_size INT,
  available_dates JSONB,
  images JSONB,
  rating DECIMAL(3,2),
  created_at TIMESTAMP
);

-- School Routes
CREATE TABLE school_routes (
  id UUID PRIMARY KEY,
  school_id UUID,
  school_name VARCHAR(200),
  route_name VARCHAR(100),
  pickup_locations JSONB,
  dropoff_location JSONB,
  departure_time TIME,
  return_time TIME,
  monthly_fee DECIMAL(10,2),
  num_children INT,
  contact_person VARCHAR(100),
  contact_phone VARCHAR(20),
  active BOOLEAN DEFAULT true,
  created_at TIMESTAMP
);

-- Ambulance Availability
CREATE TABLE ambulance_stations (
  id UUID PRIMARY KEY,
  driver_id UUID,
  location POINT,
  hospital_id UUID,
  equipment JSONB,
  driver_paramedic_trained BOOLEAN,
  available BOOLEAN DEFAULT true,
  response_time_minutes INT,
  created_at TIMESTAMP
);

-- Driver Skills & Certifications
CREATE TABLE driver_certifications (
  id UUID PRIMARY KEY,
  driver_id UUID,
  certification_type ENUM(
    'auto_rickshaw', 'taxi', 'traveller', 'truck', 
    'goods_transport', 'corporate', 'ambulance_paramedic',
    'school_transport', 'luxury_driving'
  ),
  issued_by VARCHAR(100),
  valid_until DATE,
  verified BOOLEAN,
  document_url VARCHAR(500),
  created_at TIMESTAMP
);

-- Vehicle Type Availability Per Driver
CREATE TABLE driver_vehicle_types (
  id UUID PRIMARY KEY,
  driver_id UUID,
  vehicle_type_id UUID REFERENCES vehicle_types,
  owns_vehicle BOOLEAN,
  vehicle_registration VARCHAR(20),
  insurance_valid_until DATE,
  certification_valid_until DATE,
  active BOOLEAN DEFAULT true,
  created_at TIMESTAMP
);
```

// Vehicle Management API
export const vehicleTypesAPI = {

    axiosInstance.post(`/api/corporate/approvals`, { booking_id: bookingId, contract_id: contractId }),

export const schoolTransportAPI = {
